ПРАКТИЧЕСКАЯ РАБОТА №9 — Галерея + Музыка

Цель работы

Разработать мобильное приложение на Flutter, включающее:

  • Галерею фотографий и видео в виде ленты (Instagram-стиль) с перемоткой видео
  • Раздел «Музыка» с загрузкой аудио из файлов и по URL, воспроизведением, паузой и перемоткой
  • Хранение данных в базе Hive
  • Добавление геопозиции при сохранении фотографий (пакет geolocator)

Файл models/media_item.dart — модель медиа с Hive-адаптером

import 'package:hive/hive.dart'; enum MediaType { photo, video } class MediaItem { final String id; final String path; final MediaType type; final DateTime createdAt; final double? latitude; final double? longitude; final String? locationName; MediaItem({ required this.id, required this.path, required this.type, required this.createdAt, this.latitude, this.longitude, this.locationName, }); Map<String, dynamic> toMap() => { 'id': id, 'path': path, 'type': type.index, 'createdAt': createdAt.toIso8601String(), 'latitude': latitude, 'longitude': longitude, 'locationName': locationName, }; factory MediaItem.fromMap(Map map) => MediaItem( id: map['id'], path: map['path'], type: MediaType.values[map['type']], createdAt: DateTime.parse(map['createdAt']), latitude: map['latitude'], longitude: map['longitude'], locationName: map['locationName'], ); } class MediaItemAdapter extends TypeAdapter<MediaItem> { @override final int typeId = 0; @override MediaItem read(BinaryReader r) => MediaItem.fromMap(r.readMap()); @override void write(BinaryWriter w, MediaItem o) => w.writeMap(o.toMap()); }

Файл models/audio_item.dart — модель аудио с Hive-адаптером

import 'package:hive/hive.dart'; enum AudioOrigin { file, url } class AudioItem { final String id; final String title; final String path; // file path or URL final AudioOrigin source; final DateTime addedAt; AudioItem({ required this.id, required this.title, required this.path, required this.source, required this.addedAt, }); Map<String, dynamic> toMap() => { 'id': id, 'title': title, 'path': path, 'source': source.index, 'addedAt': addedAt.toIso8601String(), }; factory AudioItem.fromMap(Map map) => AudioItem( id: map['id'], title: map['title'], path: map['path'], source: AudioOrigin.values[map['source']], addedAt: DateTime.parse(map['addedAt']), ); } class AudioItemAdapter extends TypeAdapter<AudioItem> { @override final int typeId = 1; @override AudioItem read(BinaryReader r) => AudioItem.fromMap(r.readMap()); @override void write(BinaryWriter w, AudioItem o) => w.writeMap(o.toMap()); }

Файл services/database_service.dart — сервис базы данных Hive

import 'package:hive_flutter/hive_flutter.dart'; import '../models/media_item.dart'; import '../models/audio_item.dart'; class DatabaseService { static const String mediaBoxName = 'media_items'; static const String audioBoxName = 'audio_items'; static Future<void> init() async { await Hive.initFlutter(); Hive.registerAdapter(MediaItemAdapter()); Hive.registerAdapter(AudioItemAdapter()); await Hive.openBox<MediaItem>(mediaBoxName); await Hive.openBox<AudioItem>(audioBoxName); } // Media CRUD static Box<MediaItem> get mediaBox => Hive.box(mediaBoxName); static Future<void> addMedia(MediaItem item) => mediaBox.put(item.id, item); static List<MediaItem> getAllMedia() { final items = mediaBox.values.toList(); items.sort((a, b) => b.createdAt.compareTo(a.createdAt)); return items; } static Future<void> deleteMedia(String id) => mediaBox.delete(id); // Audio CRUD static Box<AudioItem> get audioBox => Hive.box(audioBoxName); static Future<void> addAudio(AudioItem item) => audioBox.put(item.id, item); static List<AudioItem> getAllAudio() { final items = audioBox.values.toList(); items.sort((a, b) => b.addedAt.compareTo(a.addedAt)); return items; } static Future<void> deleteAudio(String id) => audioBox.delete(id); }

Файл services/location_service.dart — сервис геолокации

import 'package:geolocator/geolocator.dart'; import 'package:geocoding/geocoding.dart'; class LocationService { static Future<Position?> getCurrentPosition() async { bool serviceEnabled = await Geolocator.isLocationServiceEnabled(); if (!serviceEnabled) return null; LocationPermission permission = await Geolocator.checkPermission(); if (permission == LocationPermission.denied) { permission = await Geolocator.requestPermission(); if (permission == LocationPermission.denied) return null; } if (permission == LocationPermission.deniedForever) return null; return await Geolocator.getCurrentPosition( locationSettings: const LocationSettings(accuracy: LocationAccuracy.high), ); } static Future<String?> getLocationName(double lat, double lng) async { try { final placemarks = await placemarkFromCoordinates(lat, lng); if (placemarks.isNotEmpty) { final p = placemarks.first; final parts = [ if (p.locality?.isNotEmpty == true) p.locality!, if (p.country?.isNotEmpty == true) p.country!, ]; return parts.isNotEmpty ? parts.join(', ') : null; } } catch (_) {} return null; } }

Вывод

В ходе практической работы было разработано мобильное приложение «Gallery & Music» на Flutter, реализующее все требования на оценку 5:

  • Галерея фотографий и видео в формате ленты (Instagram-стиль) с полноценным видеоплеером Chewie (перемотка, полный экран, play/pause)
  • Раздел «Музыка» с возможностью загрузки аудиофайлов с устройства и добавления URL-ссылок на аудио, воспроизведением через just_audio с поддержкой паузы и перемотки
  • Все данные (медиа и аудио) сохраняются в локальной базе данных Hive с кастомными TypeAdapter
  • При сохранении фотографий автоматически определяется геопозиция через пакет geolocator, а название места — через geocoding

Приложение использует Material Design 3, NavigationBar для переключения между разделами и следует принципам чистой архитектуры с разделением на модели, сервисы, экраны и виджеты.
